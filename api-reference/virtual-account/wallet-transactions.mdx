---
title: 'Fetch Wallet Transactions'
api: 'GET /api/v1/wallets/fiat/{walletId}/transactions'
description: 'Retrieve all transactions for a wallet with pagination'
---

## Path Parameters

<ParamField path="walletId" type="string" required>
  The unique identifier of the wallet
</ParamField>

## Query Parameters

<ParamField query="page" type="number" default="1">
  Page number for pagination
</ParamField>

<ParamField query="limit" type="number" default="10">
  Number of transactions per page (max 100)
</ParamField>

<ParamField query="transactionType" type="string">
  Filter by transaction type: `deposit` or `withdrawal`
</ParamField>

<ParamField query="startDate" type="string">
  Start date filter (YYYY-MM-DD format)
</ParamField>

<ParamField query="endDate" type="string">
  End date filter (YYYY-MM-DD format)
</ParamField>

## Request

<ParamField header="x-api-key" type="string" required>
  API Key for authentication
</ParamField>

## Request Example

```bash
curl "https://api-staging.fossapay.com/api/v1/wallets/fiat/wal_abc123xyz/transactions?page=1&limit=10&transactionType=deposit" \
  -H "x-api-key: fp_live_sk_xxxxxxxx"
```

```javascript
const response = await fetch('https://api-staging.fossapay.com/api/v1/wallets/fiat/wal_abc123xyz/transactions?page=1&limit=10&transactionType=deposit&startDate=2024-01-01&endDate=2024-01-31', {
  headers: {
    'x-api-key': 'fp_live_sk_xxxxxxxx'
  }
});

const data = await response.json();

```

```python
import requests

url = "https://api-staging.fossapay.com/api/v1/wallets/fiat/wal_abc123xyz/transactions"
params = {
    "page": 1,
    "limit": 10,
    "transactionType": "deposit",
    "startDate": "2024-01-01",
    "endDate": "2024-01-31"

response = requests.get(url, headers=headers, params=params)
data = response.json()
transactions = data["transactions"]
```

## Pagination Example

```javascript
// Get all transactions with pagination
let page = 1;
let allTransactions = [];

while (true) {
  const response = await fetch(`https://api-staging.fossapay.com/api/v1/wallets/fiat/wal_abc123xyz/transactions?page=${page}&limit=50`, {
    headers: {
      'x-api-key': 'fp_live_sk_xxxxxxxx'
    }
  });

  const data = await response.json();
  allTransactions.push(...data.transactions);

  if (page >= data.totalPages) {
    break;
  }

  page++;
}

console.log(`Retrieved ${allTransactions.length} transactions`);
```

  For better performance, use specific date ranges and transaction type filters when possible.
